reactivemongo.api

DB

case class DB(dbName: String, connection: MongoConnection, failoverStrategy: FailoverStrategy, context: ExecutionContext) extends Product with Serializable

A Mongo Database.

Example:

import reactivemongo.api._

val connection = MongoConnection( List( "localhost:27016" ) )
val db = connection("plugin")
val collection = db("acoll")

// more explicit way
val db2 = connection.db("plugin")
val collection2 = db2.collection("plugin")
